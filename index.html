    <script>

        const canvas = document.getElementById('background-canvas');

        const ctx = canvas.getContext('2d');

        const cursorRing = document.getElementById('cursor-ring');

        

        let width, height;

        let mouseX = -1000;

        let mouseY = -1000;

        

        // オブジェクト管理

        const shards = []; // 背景の浮遊するカケラ

        const particles = []; // クリック時の爆発パーティクル

        const shardCount = 60;


        // --- クラス定義: 背景のカケラ ---

        class Shard {

            constructor() {

                this.x = Math.random() * width;

                this.y = Math.random() * height;

                this.size = Math.random() * 30 + 5; 

                this.rotation = Math.random() * Math.PI * 2;

                this.color = Math.random() > 0.85 ? '#ffffff' : '#ffff00'; 

                this.points = [];

                this.speedX = (Math.random() - 0.5) * 0.2; // ゆったり動く

                this.speedY = (Math.random() - 0.5) * 0.2;

                

                const sides = Math.floor(Math.random() * 3) + 3; 

                for (let i = 0; i < sides; i++) {

                    const angle = (i / sides) * Math.PI * 2;

                    const r = this.size * (0.5 + Math.random() * 0.5);

                    this.points.push({

                        x: Math.cos(angle) * r,

                        y: Math.sin(angle) * r

                    });

                }

            }


            update() {

                this.x += this.speedX;

                this.y += this.speedY;

                // 画面外に出たら戻す

                if(this.x < -50) this.x = width + 50;

                if(this.x > width + 50) this.x = -50;

                if(this.y < -50) this.y = height + 50;

                if(this.y > height + 50) this.y = -50;

            }


            draw(context) {

                context.save();

                context.translate(this.x, this.y);

                context.rotate(this.rotation);

                context.beginPath();

                context.moveTo(this.points[0].x, this.points[0].y);

                for (let i = 1; i < this.points.length; i++) {

                    context.lineTo(this.points[i].x, this.points[i].y);

                }

                context.closePath();

                context.fillStyle = this.color;

                context.fill();

                context.restore();

            }

        }


        // --- クラス定義: クリック時のパーティクル ---

        class Particle {

            constructor(x, y) {

                this.x = x;

                this.y = y;

                this.size = Math.random() * 5 + 1;

                this.speedX = (Math.random() - 0.5) * 10;

                this.speedY = (Math.random() - 0.5) * 10;

                this.life = 1.0; // 寿命

                this.color = Math.random() > 0.5 ? '#ffff00' : '#ffffff';

            }


            update() {

                this.x += this.speedX;

                this.y += this.speedY;

                this.life -= 0.02; // 徐々に消える

                this.speedX *= 0.95; // 摩擦

                this.speedY *= 0.95;

            }


            draw(context) {

                context.save();

                context.globalAlpha = this.life;

                context.fillStyle = this.color;

                context.beginPath();

                context.arc(this.x, this.y, this.size, 0, Math.PI * 2);

                context.fill();

                context.restore();

            }

        }


        // --- 初期化 & リサイズ ---

        function resize() {

            width = window.innerWidth;

            height = window.innerHeight;

            canvas.width = width;

            canvas.height = height;

            

            shards.length = 0;

            for(let i=0; i<shardCount; i++) {

                shards.push(new Shard());

            }

        }


        // --- アニメーションループ ---

        function animate() {

            ctx.fillStyle = '#1a1a1a';

            ctx.fillRect(0, 0, width, height);


            // 1. 背景カケラの描画（スポットライト処理）

            shards.forEach(shard => {

                shard.update(); // 動かす

                

                const dx = shard.x - mouseX;

                const dy = shard.y - mouseY;

                const dist = Math.sqrt(dx*dx + dy*dy);

                

                // スポットライト効果

                const visibleRadius = 300;

                let alpha = 0;

                if (dist < visibleRadius) {

                    alpha = 1 - (dist / visibleRadius);

                }

                alpha *= (0.8 + Math.random() * 0.2); // ゆらぎ


                if (alpha > 0) {

                    ctx.globalAlpha = alpha;

                    shard.draw(ctx);

                }

            });


            // 2. 爆発パーティクルの描画（常に光る）

            for (let i = particles.length - 1; i >= 0; i--) {

                const p = particles[i];

                p.update();

                p.draw(ctx);

                if (p.life <= 0) {

                    particles.splice(i, 1);

                }

            }


            requestAnimationFrame(animate);

        }


        // --- インタラクション ---

        function triggerExplosion(x, y) {

            // 光の爆発を生成

            for (let i = 0; i < 30; i++) {

                particles.push(new Particle(x, y));

            }

        }


        window.addEventListener('resize', resize);

        

        // マウス移動

        window.addEventListener('mousemove', (e) => {

            mouseX = e.clientX;

            mouseY = e.clientY;

            

            // カスタムカーソル追従

            cursorRing.style.left = mouseX + 'px';

            cursorRing.style.top = mouseY + 'px';

        });


        // クリックイベント

        window.addEventListener('mousedown', (e) => {

            document.body.classList.add('is-clicking');

            triggerExplosion(e.clientX, e.clientY);

        });


        window.addEventListener('mouseup', () => {

            document.body.classList.remove('is-clicking');

        });


        // タッチデバイス用

        window.addEventListener('touchmove', (e) => {

            mouseX = e.touches[0].clientX;

            mouseY = e.touches[0].clientY;

        }, {passive: true});

        

        window.addEventListener('touchstart', (e) => {

            const tx = e.touches[0].clientX;

            const ty = e.touches[0].clientY;

            mouseX = tx;

            mouseY = ty;

            triggerExplosion(tx, ty);

        }, {passive: true});


        // 実行

        resize();

        animate();


    </script>
